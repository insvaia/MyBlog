## 流式接口的调用

```js
async function fetchStream(url, data, onChunk, onComplete, onError){
    // 中止的请求控制器
    const controller = new AbortController()
    // 发起一个请求
    const response = await fetch(`http://.../${url}`,{
        methods: "POST",
        headers: {
            "content-Type": "application/json"
		}
        body: JSON.stringify(data),
        signal: controller.signal // controller中有一个signal属性
    })
    
    // 获取响应体的可读流读取器
    const reader = response.body.getReader()
    // 将二进制流数据解码为文本
    const decoder = new TextDecoder()
    
    // 流数据解码
    while(true){
        const {done, value} = await reader.read()
        if(done) break
        const chunk = decoder.decode(value, {stream: true})
        const lines = chunk.split("/n").filter(line => line.trim()) 
        for(const line of lines){
            if(line.startWith("data: ")){
                const jsonStr = line.subString(6)
                const jsonData = JSON.parse(jsonStr)
                // 返回每一块数据
                if(jsonData.type === 'chunk'){
                    onChunk(jsonData.content)
                }else if(jsonData.done){
                    onComplete()
                }else if(jsonData.error){
                    onError(jsonData.error)
                }
			}
        }
    }
    controller.abort()
}
```

## 流式接口的使用

```js
const message = ref([])
const isStreaming = ref(false)

// 获取ai响应
const getAIresponse = (userMessage) =>{
    isStream.value = true
    message.value.push({
        id: Date.now() + 1,
        role: "ai",
        content: "",
        timestamp: new Date().toISOString()
    })
    let fullResponse = ''
    fetchStream(
        "chat", 
        { message: userMessage }, 
        (chunk) =>{
        	fullResponse += chunk
        	const lastMsg = message.value[message.value.length - 1]
        	if(lastMsg && lastMsg,role === "ai"){
            	lastMsg.content = fullResponse
        	}
        	scrollToButtom()
    	},
       	() =>{
        	// AI返回完成
            isStreaming.value = false
            scrollToButtom()
     	},
        (errMsg) =>{
            const lastMsg = message.value[message.value.length - 1]
            if(lastMsg && lastMsg.role === "ai"){
                lastMsg.content = `抱歉，ai发生错误: ${errMsg}`
            }
            isStreaming.value = false
            showToast("ai回复失败")
            scrollToButtom()
        }
    )
}

```
